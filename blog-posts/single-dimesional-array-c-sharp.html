Single Dimensional Arrays in c#

Note 1: Declaration and Initialization


// Declaration and initialization of an array of integers
int[] numbers = new int[5]; // Creates an array of size 5

// Initialization with values
int[] primes = { 2, 3, 5, 7, 11 };

Note 2: Accessing Elements


int thirdPrime = primes[2]; // Accessing the 3rd element (5)

Note 3: Length Property


int size = primes.Length; // size will be 5

Note 4: Iterating Through an Array


for (int i = 0; i < primes.Length; i++)
{
    Console.WriteLine(primes[i]);
}

// Using foreach loop
foreach (int prime in primes)
{
    Console.WriteLine(prime);
}

Note 5: Sorting


System.Array.Sort(primes); // Sorts the array in ascending order

Note 6: Searching


int index = System.Array.BinarySearch(primes, 5); // Searches for 5 in the sorted array

Note 7: Filtering


var evenPrimes = primes.Where(p => p % 2 == 0); // Filters even prime numbers
foreach (int prime in evenPrimes)
{
    Console.WriteLine(prime);
}

Note 8: Finding Minimum or Maximum


int minValue = primes.Min();
int maxValue = primes.Max();

Note 9: Calculating Sum or Average


int sum = primes.Sum();
double average = primes.Average();

Note 10: Converting to String


string primesString = string.Join(", ", primes); // Converts array to string separated by comma and space
Console.WriteLine(primesString);

Note 11: Copying Arrays


int[] copyOfPrimes1 = new int[primes.Length];
System.Array.Copy(primes, copyOfPrimes1, primes.Length);

int[] copyOfPrimes2 = primes.ToArray(); // Using LINQ
foreach (var item in copyOfPrimes2)
{
    Console.WriteLine(item);
}

Note 12: Reversing the Array


System.Array.Reverse(primes); // Reverses the order of elements in the array
foreach (var item in primes)
{
    Console.WriteLine(item);
}

Note 13: Converting Array to List


List primeList = primes.ToList(); // Converts the array to a List
foreach (var item in primeList)
{
    Console.WriteLine(item);
}

Note 14: Converting List to Array


int[] newArray = primeList.ToArray(); // Converts the List back to an array
foreach (var item in newArray)
{
    Console.WriteLine(item);
}

Note 15: Resize and Remove an Item from the Array


// Find the index of the element to remove
int indexToRemove = System.Array.IndexOf(primes, 7);

// Check if the element exists in the array
if (indexToRemove != -1)
{
    // Create a new array with size one less than the original array
    int[] resizedPrimes1 = new int[primes.Length - 1];

    // Copy elements from the original array to the resized array, excluding the element to remove
    for (int i = 0, j = 0; i < primes.Length; i++)
    {
        if (i != indexToRemove)
        {
            resizedPrimes1[j] = primes[i];
            j++;
        }
    }

    // Update the reference to the new array
    primes = resizedPrimes1;
}

// Now primes array will contain { 2, 3, 5, 11 }
foreach (int prime in primes)
{
    Console.WriteLine(prime);
}

Note 16: Resize and Add an Item to the Array


// Define the new element to add
int newElement = 13;

// Create a new array with size one greater than the original array
int[] resizedPrimes2 = new int[primes.Length + 1];

// Copy elements from the original array to the resized array
for (int i = 0; i < primes.Length; i++)
{
    resizedPrimes2[i] = primes[i];
}

// Add the new element at the end of the resized array
resizedPrimes2[resizedPrimes2.Length - 1] = newElement;

// Update the reference to the new array
primes = resizedPrimes2;

// Now primes array will contain { 2, 3, 5, 7, 11, 13 }
foreach (int prime in primes)
{
    Console.WriteLine(prime);
}

Note 17: Clearing the Array


System.Array.Clear(primes, 0, primes.Length); // Clears all elements of the array
